Step 1: Define Tools

Tools are functions the agent can call. Each tool needs a name, description, and parameters — the LLM uses these to decide when to use each tool.

def web_search(query: str) -> str:

    # In production, use: SerpAPI, Perplexity API, or Brave Search API
    return f"[Search results for '{query}']: This is a simulated response. Integrate a real search API for production use."

def read_file(path: str) -> str:
    try:
        with open(path, 'r') as f:
            return f.read()
    except FileNotFoundError:
        return f"Error: File '{path}' not found."
    except Exception as e:
        return f"Error reading file: {str(e)}"

def write_file(path: str, content: str) -> str:
    try:
        os.makedirs(os.path.dirname(path) if os.path.dirname(path) else '.', exist_ok=True)
        with open(path, 'w') as f:
            f.write(content)
        return f"Successfully wrote {len(content)} characters to {path}"
    except Exception as e:
        return f"Error writing file: {str(e)}"

def run_python(code: str) -> str:
    try:
        # WARNING: In production, use a sandboxed environment
        result = subprocess.run(
            ['python3', '-c', code],
            capture_output=True, text=True, timeout=30
        )
        if result.returncode == 0:
            return result.stdout.strip() or "Code executed successfully (no output)"
        else:
            return f"Error: {result.stderr.strip()}"
    except subprocess.TimeoutExpired:
        return "Error: Code execution timed out (30s limit)"
    except Exception as e:
        return f"Error executing code: {str(e)}"

def get_datetime() -> str:
    return datetime.now().strftime("%Y-%m-%d %H:%M:%S UTC")

# Tool dispatcher
TOOL_FUNCTIONS = {
    "web_search": web_search,
    "read_file": read_file,
    "write_file": write_file,
    "run_python": run_python,
    "get_datetime": get_datetime,
}

def execute_tool(tool_name: str, arguments: dict) -> str:
    """Execute a tool by name with given arguments."""
    if tool_name not in TOOL_FUNCTIONS:
        return f"Error: Unknown tool '{tool_name}'"
    try:
        return TOOL_FUNCTIONS[tool_name](**arguments)
    except TypeError as e:
        return f"Error: Invalid arguments for {tool_name}: {str(e)}"
    except Exception as e:
        return f"Error executing {tool_name}: {str(e)}"

Step 2: Add Memory

Without memory, the agent forgets everything between messages. We implement a simple sliding window + summary approach.

Code: memory.py

from collections import deque
from typing import List, Dict

class AgentMemory:
    """Sliding window memory with automatic summarization."""
    
    def __init__(self, max_messages: int = 20):
        self.max_messages = max_messages
        self.messages: deque = deque(maxlen=max_messages)
        self.summaries: List[str] = []
        self.tool_results: List[Dict] = []
    
    def add_message(self, role: str, content: str, tool_calls=None):
        """Add a message to memory."""
        msg = {"role": role, "content": content}
        if tool_calls:
            msg["tool_calls"] = tool_calls
        self.messages.append(msg)
    
    def add_tool_result(self, tool_name: str, result: str):
        """Store tool result for context."""
        self.tool_results.append({
            "tool": tool_name,
            "result": result[:500]  # Truncate long results
        })
        # Keep only last 10 tool results
        if len(self.tool_results) > 10:
            self.tool_results = self.tool_results[-10:]
    
    def get_context(self) -> List[Dict]:
        """Get full conversation context for the LLM."""
        context = []
        
        # Add summary of older conversation if exists
        if self.summaries:
            context.append({
                "role": "system",
                "content": f"Previous conversation summary: {' | '.join(self.summaries)}"
            })
        
        # Add recent messages
        context.extend(list(self.messages))
        
        return context
    
    def summarize(self, llm_client, model: str = "gpt-4o-mini"):
        """Summarize older messages to free up context window."""
        if len(self.messages) < 10:
            return
        
        # Take first half of messages to summarize
        to_summarize = list(self.messages)[:len(self.messages)//2]
        
        summary_prompt = "Summarize this conversation concisely, preserving key facts, decisions, and context:\n\n"
        for msg in to_summarize:
            summary_prompt += f"{msg['role']}: {msg['content'][:200]}\n"
        
        try:
            response = llm_client.chat.completions.create(
                model=model,
                messages=[{"role": "user", "content": summary_prompt}],
                max_tokens=200
            )
            summary = response.choices[0].message.content
            self.summaries.append(summary)
            
            # Remove summarized messages
            for _ in range(len(to_summarize)):
                if self.messages:
                    self.messages.popleft()
                    
        except Exception as e:
            print(f"Summarization failed: {e}")

Step 3: Build the Agent Loop

The agent loop is the core. It sends the user's message to the LLM, checks if the LLM wants to use a tool, executes the tool, feeds the result back, and repeats until the LLM has a final answer.

Code: agent.py

import os
import json
from openai import OpenAI
from dotenv import load_dotenv
from tools import TOOLS, execute_tool
from memory import AgentMemory

load_dotenv()

class Agent:
    def __init__(self, model: str = "gpt-4o-mini", max_iterations: int = 10):
        self.client = OpenAI(api_key=os.getenv("OPENAI_API_KEY"))
        self.model = model
        self.max_iterations = max_iterations
        self.memory = AgentMemory()
        
        # System prompt — this is where you define the agent's personality and rules
        self.system_prompt = """You are a helpful AI assistant with access to tools.
You can search the web, read/write files, execute Python code, and check the date/time.

Rules:
1. Always think step by step before using a tool
2. If you need information, use the appropriate tool
3. If a tool fails, try a different approach
4. Be concise but thorough
5. When writing files, confirm the path before writing
6. Never execute destructive commands without asking

When you have a final answer, respond directly without calling any tools."""
    
    def run(self, user_message: str) -> str:
        """Run the agent on a user message."""
        self.memory.add_message("user", user_message)
        
        for iteration in range(self.max_iterations):
            # Build messages for LLM
            messages = [{"role": "system", "content": self.system_prompt}]
            messages.extend(self.memory.get_context())
            
            # Call LLM
            response = self.client.chat.completions.create(
                model=self.model,
                messages=messages,
                tools=TOOLS,
                tool_choice="auto"
            )
            
            message = response.choices[0].message
            
            # Check if LLM wants to use tools
            if message.tool_calls:
                # Add assistant message with tool calls
                self.memory.add_message(
                    "assistant",
                    message.content or "",
                    tool_calls=[{
                        "id": tc.id,
                        "type": "function",
                        "function": {"name": tc.function.name, "arguments": tc.function.arguments}
                    } for tc in message.tool_calls]
                )
                
                # Execute each tool
                for tool_call in message.tool_calls:
                    tool_name = tool_call.function.name
                    arguments = json.loads(tool_call.function.arguments)
                    
                    print(f"  🔧 Using tool: {tool_name}({arguments})")
                    
                    result = execute_tool(tool_name, arguments)
                    self.memory.add_tool_result(tool_name, result)
                    
                    # Add tool result to conversation
                    self.memory.add_message(
                        "tool",
                        result,
                        tool_call_id=tool_call.id
                    )
                
                # Continue loop — LLM will see tool results and respond
                continue
            
            else:
                # No tool calls — this is the final answer
                final_response = message.content
                self.memory.add_message("assistant", final_response)
                return final_response
        
        # Max iterations reached
        return "I've reached my iteration limit. Here's what I found so far. Could you clarify what else you need?"

# Usage
if __name__ == "__main__":
    agent = Agent()
    
    print("🤖 AI Agent initialized. Type 'quit' to exit.\n")
    
    while True:
        user_input = input("You: ").strip()
        if user_input.lower() in ['quit', 'exit', 'q']:
            print("Goodbye! 👋")
            break
        
        print("Agent:", end=" ")
        response = agent.run(user_input)
        print(response)
        print()

Step 4: Error Handling

The agent above works, but production agents need robust error handling. Here's the enhanced version:

def run_safe(self, user_message: str) -> dict:
    """Run agent with full error handling. Returns dict with response and metadata."""
    start_time = datetime.now()
    
    try:
        if not user_message or not user_message.strip():
            return {"success": False, "error": "Empty message", "response": "Please enter a message."}
        
        if len(user_message) > 10000:
            return {"success": False, "error": "Message too long", "response": "Message exceeds 10,000 character limit."}
        
        response = self.run(user_message)
        
        elapsed = (datetime.now() - start_time).total_seconds()
        
        return {
            "success": True,
            "response": response,
            "elapsed_seconds": elapsed,
            "iterations": self.max_iterations,
            "memory_size": len(self.memory.messages)
        }
        
    except Exception as e:
        return {
            "success": False,
            "error": str(e),
            "response": f"An error occurred: {str(e)}. Please try again."
        }
